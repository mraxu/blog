# windows msc命令整理
```
# 证书管理实用程序
certmgr.msc

# 本地安全策略
secpol.msc

# 计算机管理
compmgmt.msc

# 设备管理器
devmgmt.msc

# 磁盘管理程序
dfrg.msc

# 共享文件夹管理
fsmgmt.msc

# 组策略
gpedit.msc

# 本地用户和组
lusrmgr.msc

# 计算机性能检测程序
perfmon.msc

# 本地服务设置
services.msc

# 启动计算器
calc

# 垃圾整理
cleanmgr

# 事件查看器
eventvwr

# 打开`网络和共享`
ncpa.cpl
```

# 功能命令
```
# 注销
logoff

# 关机
shutdown -p

# 重启
shutdown -r -t 0
```
